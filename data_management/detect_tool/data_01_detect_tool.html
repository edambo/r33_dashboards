Load data

An API token unique to each user and project can be requested on REDCap. To request a token, you can follow the instructions on the repository Wiki https://github.com/brad-cannell/r33_dashboards/wiki/DETECT%E2%80%90RPC-Data#redcap This API token can be saved on your system using the keyring key_set function as follows: keyring::key_set(“detect_tool_redcap_api”). A password box will pop-up and your unique token can be entered into the text box. Once this is done once, it doesn’t need to be repeated unless a new token is requested on REDCap. The code below can then be run.

# Define REDCap API URL and API token
api_token <-  keyring::key_get("detect_tool_redcap_api") # Get saved REDCap API token
redcap_url <- "https://redcap.uth.tmc.edu/api/"

# Establish connection to the REDCap project
rcon <- redcapConnection(url = redcap_url, token = api_token)

# Export records with coded values
tool_raw  <- exportRecordsTyped(rcon, form = "reporting_instrument",
                                rawOrLabel = "raw", factor = FALSE)

An API token can be requested on GO UTHealth. To request a token, you can follow the instructions on the repository Wiki https://github.com/brad-cannell/r33_dashboards/wiki/DETECT%E2%80%90RPC-Data#go-ut-health This API token along with the client application name can be saved on your system using the keyring key_set function with the service argument = “detect_tool_go_uth_api”. The code below can then be run.

go_uth_app_name <- keyring::key_list("detect_tool_go_uth_api")[1,2]
go_uth_api_token <-  keyring::key_get("detect_tool_go_uth_api", go_uth_app_name) # Get saved API token

## Function to get link data
go_uth_link_api <- function(slug){

  go_uth_url <- paste0("https://apps.uth.edu/go/api/urls/", slug, "/traffic")
  
  start_date <- "03/12/2025" # Date of first launch of DETECT-RPC TOOL
  end_date <- format(Sys.Date(), format = "%m/%d/%Y") # today's date
  
  details <- POST(go_uth_url, 
                  add_headers(
                    `Authorization` = paste("Bearer", go_uth_api_token),
                    `Content-type` = "application/json",
                    `appName` = go_uth_app_name,
                    `startDate` = start_date,
                    `endDate` = end_date
                  ))
  
  # Extract and parse the JSON response
  raw_content <- content(details, "text", encoding = "UTF-8")
  parsed_json <- fromJSON(raw_content, flatten = TRUE)
  
  # Create column name based on slug
  new_col_name <- snakecase::to_snake_case(slug)
  # Extract the traffic data frame from parsed_json list
  parsed_json$traffic %>% rename_with(~new_col_name, "hit") 
}

Create new variables

Replace reporting instument timestamp variable with timestamp end variable

The reporting_instrument_timestamp variable contains information on both the submission time and the completion status of each survey response. The variable reporting_instrument_complete also provides completion status information. A new variable will be created that only contains the sumbmission time and the reporting_instrument_timestamp variable will be removed.

tool <- tool_raw %>% 
  mutate(
    timestamp_end = case_when(
      is.na(reporting_instrument_timestamp) ~ NA,
      reporting_instrument_timestamp == "[not completed]" ~ NA,
      TRUE ~ reporting_instrument_timestamp
    ),
    timestamp_end = as.POSIXct(timestamp_end, tz = "UTC")
  ) %>% relocate(c(timestamp_end, reporting_instrument_complete),
                 .after = ri_timestamp_start) %>%
  select(-c(reporting_instrument_timestamp))

tool_vars <- names(tool) %>% 
  # Subset new variables
  setdiff(., var_desc$variable) %>%
  list() %>%
  # Convert list of new variables into dataframe
  as.data.frame(col.names = "variable") %>%
  # Join descriptions from var_desc dataframe to new data frame 
  mutate(
    no_suffix = gsub("_[0-9]+cat[_f]*", "", x = variable)
  ) %>% inner_join(var_desc, by = c("no_suffix" = "variable")) %>%
  # remove "no_suffix" variable
  select(-c(no_suffix))

Merge link data and convert to long format

# Merge all the link data into one data frame
link_hits <- list(definitions_link_hits, al_link_hits, ca_link_hits, 
                  md_link_hits, tx_link_hits) %>% purrr::reduce(left_join, 
                                                                by = "date") %>%
  # Convert data frame to long form
  tidyr::pivot_longer(!date, names_to = "link", values_to = "hits") %>%
  mutate(
    date = lubridate::mdy(date)
  )
